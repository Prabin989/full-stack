let calcDisplay = '0';
let calcPrevious = null;
let calcOperation = null;
let calcWaitingForOperand = false;

function inputNumber(num) {
    if (calcWaitingForOperand) {
        calcDisplay = num;
        calcWaitingForOperand = false;
    } else {
        calcDisplay = calcDisplay === '0' ? num : calcDisplay + num;
    }
    updateCalcDisplay();
}

function inputOperator(nextOperator) {
    const inputValue = parseFloat(calcDisplay);
    
    if (calcPrevious === null) {
        calcPrevious = inputValue;
    } else if (calcOperation) {
        const result = performCalculation[calcOperation](calcPrevious, inputValue);
        calcDisplay = String(result);
        calcPrevious = result;
        updateCalcDisplay();
    }
    
    calcWaitingForOperand = true;
    calcOperation = nextOperator;
}

const performCalculation = {
    '/': (prev, next) => prev / next,
    '*': (prev, next) => prev * next,
    '+': (prev, next) => prev + next,
    '-': (prev, next) => prev - next,
};

function calculate() {
    const inputValue = parseFloat(calcDisplay);
    
    if (calcPrevious !== null && calcOperation) {
        const result = performCalculation[calcOperation](calcPrevious, inputValue);
        calcDisplay = String(result);
        calcPrevious = null;
        calcOperation = null;
        calcWaitingForOperand = true;
        updateCalcDisplay();
    }
}
